SELECT I.SUMMARY, IL.NAME
FROM ISSUES I
JOIN ISSUESLISTS IL ON I.ISSUESLIST_ID = IL.ID;

SELECT I.SUMMARY AS TASK, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U ON USER_ID_ASSIGNEDTO = U.ID;


SELECT I.USER_ID_ASSIGNEDTO AS ID, U.LASTNAME, count(*) AS TASK_QUANTITY
FROM USERS U JOIN ISSUES I
ON U.ID = I.USER_ID_ASSIGNEDTO
group by U.ID
HAVING count(*)>0;
